      // The module pattern looks familiar
      // Except that it has concepts of public and private
      var Calculator = function (eq) {
        //private member
        var eqCtl = document.getElementById(eq);

        return {
          // expose public member
          add: function (x, y) {
            var val = x + y;
            eqCtl.innerHTML += val;
          }
        };
      };
